import React, { useState, useEffect } from 'react';
import { Home, Volume2, Video, Image, FileText, ClipboardList, X, Menu } from 'lucide-react';

// Configuration for easy content management
const CONFIG = {
  logo: '/api/placeholder/100/100',
  title: 'Model Viewer Gallery',
  model: {
    src: 'https://modelviewer.dev/shared-assets/models/Astronaut.glb',
    alt: '3D Model',
    poster: 'https://modelviewer.dev/shared-assets/models/Astronaut.webp'
  },
  content: {
    home: {
      available: true,
      data: {
        title: 'Welcome to Model Viewer Gallery',
        description: 'Explore interactive 3D models with rich multimedia content.'
      }
    },
    audio: {
      available: true,
      data: [
        { title: 'Sample Audio 1', url: 'https://www.soundhelix.com/examples/mp3/SoundHelix-Song-1.mp3' },
        { title: 'Sample Audio 2', url: 'https://www.soundhelix.com/examples/mp3/SoundHelix-Song-2.mp3' }
      ]
    },
    video: {
      available: true,
      data: [
        { title: 'Sample Video', type: 'youtube', url: 'dQw4w9WgXcQ' },
        { title: 'Direct Video', type: 'mp4', url: 'https://www.w3schools.com/html/mov_bbb.mp4' }
      ]
    },
    images: {
      available: true,
      data: [
        { title: 'Image 1', url: '/api/placeholder/800/600' },
        { title: 'Image 2', url: '/api/placeholder/800/600' },
        { title: 'Image 3', url: '/api/placeholder/800/600' }
      ]
    },
    articles: {
      available: false,
      data: []
    },
    quiz: {
      available: true,
      data: {
        title: '3D Model Knowledge Quiz',
        questions: [
          {
            question: 'What does 3D modeling stand for?',
            options: ['Three Dimensional', 'Triple Display', 'Tri-Data', 'Third Degree'],
            correct: 0
          },
          {
            question: 'Which file format is commonly used for 3D models?',
            options: ['DOC', 'GLB', 'MP3', 'PDF'],
            correct: 1
          }
        ]
      }
    }
  }
};

const ModelViewerApp = () => {
  const [activePopup, setActivePopup] = useState(null);
  const [mobileMenuOpen, setMobileMenuOpen] = useState(false);
  const [quizAnswers, setQuizAnswers] = useState({});
  const [quizSubmitted, setQuizSubmitted] = useState(false);

  useEffect(() => {
    const script = document.createElement('script');
    script.type = 'module';
    script.src = 'https://ajax.googleapis.com/ajax/libs/model-viewer/4.0.0/model-viewer.min.js';
    document.head.appendChild(script);

    return () => {
      document.head.removeChild(script);
    };
  }, []);

  const navItems = [
    { id: 'home', icon: Home, label: 'Home' },
    { id: 'audio', icon: Volume2, label: 'Audio' },
    { id: 'video', icon: Video, label: 'Video' },
    { id: 'images', icon: Image, label: 'Gallery' },
    { id: 'articles', icon: FileText, label: 'Articles' },
    { id: 'quiz', icon: ClipboardList, label: 'Quiz' }
  ];

  const handleNavClick = (itemId) => {
    if (!CONFIG.content[itemId].available) {
      setActivePopup({ type: 'error', id: itemId });
    } else {
      setActivePopup({ type: itemId, data: CONFIG.content[itemId].data });
    }
    setMobileMenuOpen(false);
    setQuizAnswers({});
    setQuizSubmitted(false);
  };

  const closePopup = () => {
    setActivePopup(null);
    setQuizAnswers({});
    setQuizSubmitted(false);
  };

  const handleQuizAnswer = (questionIndex, answerIndex) => {
    setQuizAnswers(prev => ({ ...prev, [questionIndex]: answerIndex }));
  };

  const submitQuiz = () => {
    setQuizSubmitted(true);
  };

  const renderPopupContent = () => {
    if (!activePopup) return null;

    if (activePopup.type === 'error') {
      return (
        <div className="text-center py-12">
          <h2 className="text-6xl font-bold text-red-500 mb-4">404</h2>
          <p className="text-xl text-gray-700">Content Not Available</p>
        </div>
      );
    }

    switch (activePopup.type) {
      case 'home':
        return (
          <div className="py-8">
            <h2 className="text-3xl font-bold mb-4">{activePopup.data.title}</h2>
            <p className="text-lg text-gray-700">{activePopup.data.description}</p>
          </div>
        );

      case 'audio':
        return (
          <div className="space-y-6">
            <h2 className="text-2xl font-bold mb-4">Audio Content</h2>
            {activePopup.data.map((audio, idx) => (
              <div key={idx} className="bg-gray-50 p-4 rounded-lg">
                <h3 className="font-semibold mb-2">{audio.title}</h3>
                <audio controls className="w-full">
                  <source src={audio.url} type="audio/mpeg" />
                  Your browser does not support the audio element.
                </audio>
              </div>
            ))}
          </div>
        );

      case 'video':
        return (
          <div className="space-y-6">
            <h2 className="text-2xl font-bold mb-4">Video Content</h2>
            {activePopup.data.map((video, idx) => (
              <div key={idx} className="bg-gray-50 p-4 rounded-lg">
                <h3 className="font-semibold mb-2">{video.title}</h3>
                {video.type === 'youtube' ? (
                  <div className="aspect-video">
                    <iframe
                      className="w-full h-full rounded"
                      src={`https://www.youtube.com/embed/${video.url}`}
                      title={video.title}
                      frameBorder="0"
                      allow="accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
                      allowFullScreen
                    />
                  </div>
                ) : (
                  <video controls className="w-full rounded">
                    <source src={video.url} type="video/mp4" />
                    Your browser does not support the video element.
                  </video>
                )}
              </div>
            ))}
          </div>
        );

      case 'images':
        return (
          <div>
            <h2 className="text-2xl font-bold mb-4">Image Gallery</h2>
            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
              {activePopup.data.map((img, idx) => (
                <div key={idx} className="bg-gray-50 p-2 rounded-lg">
                  <img src={img.url} alt={img.title} className="w-full h-auto rounded" />
                  <p className="text-center mt-2 font-semibold">{img.title}</p>
                </div>
              ))}
            </div>
          </div>
        );

      case 'quiz':
        return (
          <div>
            <h2 className="text-2xl font-bold mb-4">{activePopup.data.title}</h2>
            <div className="space-y-6">
              {activePopup.data.questions.map((q, qIdx) => (
                <div key={qIdx} className="bg-gray-50 p-4 rounded-lg">
                  <p className="font-semibold mb-3">{qIdx + 1}. {q.question}</p>
                  <div className="space-y-2">
                    {q.options.map((opt, oIdx) => (
                      <label
                        key={oIdx}
                        className={`flex items-center p-2 rounded cursor-pointer hover:bg-gray-100 ${
                          quizSubmitted
                            ? oIdx === q.correct
                              ? 'bg-green-100'
                              : quizAnswers[qIdx] === oIdx
                              ? 'bg-red-100'
                              : ''
                            : quizAnswers[qIdx] === oIdx
                            ? 'bg-blue-100'
                            : ''
                        }`}
                      >
                        <input
                          type="radio"
                          name={`question-${qIdx}`}
                          checked={quizAnswers[qIdx] === oIdx}
                          onChange={() => handleQuizAnswer(qIdx, oIdx)}
                          disabled={quizSubmitted}
                          className="mr-3"
                        />
                        {opt}
                      </label>
                    ))}
                  </div>
                </div>
              ))}
              {!quizSubmitted ? (
                <button
                  onClick={submitQuiz}
                  className="w-full bg-blue-500 text-white py-3 rounded-lg font-semibold hover:bg-blue-600"
                >
                  Submit Quiz
                </button>
              ) : (
                <div className="bg-green-50 p-4 rounded-lg text-center">
                  <p className="text-lg font-semibold">
                    Score: {Object.keys(quizAnswers).filter(k => quizAnswers[k] === activePopup.data.questions[k].correct).length} / {activePopup.data.questions.length}
                  </p>
                </div>
              )}
            </div>
          </div>
        );

      default:
        return null;
    }
  };

  return (
    <div className="min-h-screen bg-gray-100">
      {/* Header */}
      <header className="bg-white shadow-md sticky top-0 z-40">
        <div className="container mx-auto px-4 py-3">
          <div className="flex items-center justify-between">
            <div className="flex items-center gap-4">
              <img src={CONFIG.logo} alt="Logo" className="w-12 h-12 object-cover rounded" />
              <h1 className="text-xl md:text-2xl font-bold text-gray-800">{CONFIG.title}</h1>
            </div>
            
            {/* Desktop Navigation */}
            <nav className="hidden md:flex gap-2">
              {navItems.map(item => (
                <button
                  key={item.id}
                  onClick={() => handleNavClick(item.id)}
                  className="flex items-center gap-2 px-4 py-2 rounded-lg hover:bg-gray-100 transition-colors"
                >
                  <item.icon size={18} />
                  <span>{item.label}</span>
                </button>
              ))}
            </nav>

            {/* Mobile Menu Button */}
            <button
              onClick={() => setMobileMenuOpen(!mobileMenuOpen)}
              className="md:hidden p-2 hover:bg-gray-100 rounded-lg"
            >
              <Menu size={24} />
            </button>
          </div>

          {/* Mobile Navigation */}
          {mobileMenuOpen && (
            <nav className="md:hidden mt-4 flex flex-col gap-2">
              {navItems.map(item => (
                <button
                  key={item.id}
                  onClick={() => handleNavClick(item.id)}
                  className="flex items-center gap-2 px-4 py-3 rounded-lg hover:bg-gray-100 transition-colors text-left"
                >
                  <item.icon size={18} />
                  <span>{item.label}</span>
                </button>
              ))}
            </nav>
          )}
        </div>
      </header>

      {/* Main Content */}
      <main className="container mx-auto px-4 py-8">
        <div className="bg-white rounded-lg shadow-lg p-4 md:p-8">
          <div className="model-viewer-container">
            <model-viewer
              src={CONFIG.model.src}
              alt={CONFIG.model.alt}
              poster={CONFIG.model.poster}
              camera-controls
              auto-rotate
              shadow-intensity="1"
              style={{ width: '100%', height: '500px' }}
            />
          </div>
        </div>
      </main>

      {/* Popup Modal */}
      {activePopup && (
        <div className="fixed inset-0 bg-black bg-opacity-50 z-50 flex items-center justify-center p-4">
          <div className="bg-white rounded-lg shadow-xl max-w-4xl w-full max-h-[90vh] overflow-y-auto">
            <div className="sticky top-0 bg-white border-b px-6 py-4 flex justify-between items-center">
              <h2 className="text-xl font-bold">
                {activePopup.type === 'error' ? 'Error' : navItems.find(item => item.id === activePopup.type)?.label}
              </h2>
              <button
                onClick={closePopup}
                className="p-2 hover:bg-gray-100 rounded-full transition-colors"
              >
                <X size={24} />
              </button>
            </div>
            <div className="px-6 py-4">
              {renderPopupContent()}
            </div>
          </div>
        </div>
      )}
    </div>
  );
};

export default ModelViewerApp;
